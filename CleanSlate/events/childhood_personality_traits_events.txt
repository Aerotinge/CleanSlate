# Event ID 1500-1509 is reserved for Gluttonous events

# Gain Gluttony
character_event = {
	id = 1500
	desc = EVTDESC1500
	picture = GFX_evt_child_play

	is_triggered_only = yes # on_yearly_childhood_pulse

	lacks_dlc = "Conclave"
	prisoner = no
	only_capable = yes

	trigger = {
		has_guardian = yes
		personality_traits < 5

		NOR = {
			trait = temperate
			trait = gluttonous
			has_character_flag = is_child_of_destiny
		}
	}

	weight_multiplier = {
		days = 1

		mult_modifier = {
			factor = 2
			trait = slothful
		}

		mult_modifier = {
			factor = 1.05
			trait = clubfooted
		}

		mult_modifier = {
			factor = 1.10
			trait = harelip
		}

		mult_modifier = {
			factor = 1.5
			trait = hunchback
		}

		mult_modifier = {
			factor = 1.05
			trait = lisp
		}

		mult_modifier = {
			factor = 1.05
			trait = stutter
		}

		mult_modifier = {
			factor = 0.95
			has_fair_trait_trigger = yes
		}

		mult_modifier = {
			factor = 1.10
			has_ugly_trait_trigger = yes
		}

		mult_modifier = {
			factor = 1.2
			trait = dwarf
		}

		mult_modifier = {
			factor = 0.95
			trait = genius
		}

		mult_modifier = {
			factor = 0.95

			OR = {
				trait = quick
				trait = shrewd
			}
		}

		mult_modifier = {
			factor = 1.2

			OR = {
				trait = slow
				trait = dull
			}
		}

		mult_modifier = {
			factor = 1.5
			trait = imbecile
		}

		mult_modifier = {
			factor = 1.5
			trait = inbred
		}

		mult_modifier = {
			factor = 0.05
			is_strong_trigger = yes
		}

		mult_modifier = {
			factor = 5.0
			is_weak_trigger = yes
		}
	}

	option = {
		name = EVTOPTA1500

		add_trait = gluttonous
		gain_weight_medium_effect = yes
		weight_trait_check_effect = yes

		guardian = {
			character_event = {
				id = 2000
				days = 3
			}
		}
	}
}
